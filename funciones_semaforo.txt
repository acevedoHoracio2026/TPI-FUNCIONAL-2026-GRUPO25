

;; obtener hora y fecha del sistema
;;-----------------------------------
(multiple-value-bind (segundo minuto hora fecha mes año dia-semana)
(get-decoded-time)

;;---------------------------------------------------------------------------
(defun tiempo_actual()
   (multiple-value-bind (segundo minuto hora dia mes año)(get-decoded-time)
   (list hora minuto segundo dia mes año)))
------------------------------------------------------------------------------
(defun hora ()
      (nth 0 (tiempo_actual)))
--------------------------------------------------------------------------------
(defun minuto()
      (nth 1 (tiempo_actual)))
----------------------------------------------------------------------------
(defun segundos()
    (nth 2 (tiempo_actual)))
-----------------------------------------------------------------------------
(defun mes()
    (nth 4 (tiempo_actual)))
-----------------------------------------------------------------------------------
(defun dia()
    (nth 3 (tiempo_actual)))
-----------------------------------------------------------------------------------
